    return func

    print('func run')

f = test(func)

f.__name__
→'func'

语法糖

@decorator      //相当于decorator(test)
def test():
    return random.random()

@decorator
def test_two():
    return random.random()*10

def decorator(func):
    def wrapper(\*args, \*\*kwargs):
        # do something
        return round(func(\*args, \*\*kwargs), 2)  //保留两位有效数字
    return wrapper

输入

test.__name__
→'wrapper'

test()
→0.92

类

初始化函数、模板函数……

class Person:

def __init__(self, name, age)
    self._name = name
    self._age = age

@property
def get_name(self):
    return    self.name

def rename(self, new_name):
    self._name = new_name

输入

p = Person('wong', 12)
p.get_name()
→'wong'

加入@property装饰器在函数getname() 前
输入p.get_name就行（不加括号）
下划线’\‘表明类内的属性不应该暴露给用户，是私有的

class Student(Person):
    def set_score(self, score)
        self._score = score

    def get_score(self):
        return self._score
